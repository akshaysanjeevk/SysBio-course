\section{Decision Tree}
\begin{frame}{Informal Description}
    \begin{align}
        H^i &= -\sum_j^n p(x^i_j)\log p(x^i_j)\\
        IG^i &= H^i - \sum_s \frac{|x^{is}|}{|x^{i}|} H^is
    \end{align}
    \[x_{\text{selected}} = \arg \max_{i} (IG^i)\]
\end{frame}
